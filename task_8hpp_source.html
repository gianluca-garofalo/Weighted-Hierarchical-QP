112 template<typename T>
113 struct is_reference_wrapper : std::false_type {};

115 template<typename U>
116 struct is_reference_wrapper<std::reference_wrapper<U>> : std::true_type {};
117
118 // Unwrap a reference_wrapper; otherwise, forward the value
119 template<typename T>
120 static decltype(auto) unwrap(T&& t) {
121 if constexpr (is_reference_wrapper<std::decay_t<T>>::value) {
122 return t.get();
123 } else {
124 return std::forward<T>(t);
125 }
126 }
127
128 // If T is a reference, store as reference_wrapper (to allow rebinding) otherwise, store by value
129 template<typename T>
130 using ArgStorage =
131 std::conditional_t< std::is_reference<T>::value, std::reference_wrapper<std::remove_reference_t<T>>, T>;
132
133 protected:
134 // Use an optional tuple to delay initialization.
135 std::optional<std::tuple<ArgStorage<Args>...>> args_;
136
144 virtual void run(Args... args) = 0;
145
146 void compute() override {
147 std::apply(
148 [this](auto&&... args) {
149 run(unwrap(std::forward<decltype(args)>(args))...);
150 },
151 *args_);
152
153 assert(matrix_.rows() == upper_.rows());
154 assert(lower_.rows() == upper_.rows());
155 assert(activeUpSet_.size() == upper_.rows());
156
157 activeLowSet_ = activeUpSet_ = upper_.array() == lower_.array();
158
159 if (!indices_.size()) {
160 auto n = matrix_.cols();
161 indices_ = Eigen::VectorXi::LinSpaced(n, 0, n - 1);
162 }
163 isComputed_ = true;
164 }

185 // If a parameter is a reference (or const reference), the corresponding argument must be an lvalue.
186 template<

188 typename = std::enable_if_t< (sizeof...(U) == sizeof...(Args)) &&
189 ((!std::is_reference<Args>::value || std::is_lvalue_reference<U>::value) && ...)>>
190 void update(U&&... args) {
191 args_ = std::tuple<ArgStorage<Args>...>(std::forward<U>(args)...);
192 isComputed_ = false;
