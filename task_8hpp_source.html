96 // Unwrap a reference_wrapper; otherwise, forward the value
97 template<typename T>
98 static decltype(auto) unwrap(T&& t) {
99 if constexpr (is_reference_wrapper<std::decay_t<T>>::value) {
100 return t.get();
101 } else {

103 }
104 }

124 void compute() override {
125 if (isComputed_) {
126 return;
127 }
128
129 std::apply(
130 [this](auto&&... args) {
131 run(unwrap(std::forward<decltype(args)>(args))...);
132 },
133 *args_);
134
135 assert(matrix_.rows() == upper_.rows());
136 assert(lower_.rows() == upper_.rows());
137
138 if (!indices_.size()) {
139 indices_ = Eigen::VectorXi::Ones(matrix_.cols());
140 }
141 if (weight_.size()) {
142 // Weight subtasks within task
143 matrix_ = weight_.matrixU() * matrix_;
144 lower_ = weight_.matrixU() * lower_;
145 upper_ = weight_.matrixU() * upper_;
146 }
147
148 Eigen::MatrixXd tmp = matrix_;
149 matrix_.resize(tmp.rows(), indices_.size());
150 for (int k = 0, h = 0; h < indices_.size(); ++h) {
151 if (indices_(h)) {
152 matrix_.col(h) = tmp.col(k++);
153 }
154 }
155
156 isComputed_ = true;
157 }

169 }
