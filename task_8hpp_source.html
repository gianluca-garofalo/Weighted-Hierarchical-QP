81 // Unwrap a reference_wrapper; otherwise, forward the value
82 template<typename T>
83 static decltype(auto) unwrap(T&& t) {
84 if constexpr (is_reference_wrapper<std::decay_t<T>>::value) {
85 return t.get();
86 } else {
87 return std::forward<T>(t);
88 }
89 }
90
91 // If T is a reference, store as reference_wrapper (to allow rebinding) otherwise, store by value
92 template<typename T>
93 using ArgStorage =
94 std::conditional_t< std::is_reference<T>::value, std::reference_wrapper<std::remove_reference_t<T>>, T>;
95
96 protected:
97 // Use an optional tuple to delay initialization.
98 std::optional<std::tuple<ArgStorage<Args>...>> args_;
99
107 virtual void run(Args... args) = 0;
108
109 void compute() override {
110 std::apply(
111 [this](auto&&... args) {
112 run(unwrap(std::forward<decltype(args)>(args))...);
113 },
114 *args_);
115
116 assert(matrix_.rows() == vector_.rows());
117 assert(equalitySet_.size() == vector_.rows());
118
119 if (!indices_.size()) {
120 auto n = matrix_.cols();
121 indices_ = Eigen::VectorXi::LinSpaced(n, 0, n - 1);
122 }
123 isComputed_ = true;
124 }
