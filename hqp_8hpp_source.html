118 class ConstraintTracker {
119 private:
120 std::vector<int> buffer_; // Just store constraint rows
121 std::unordered_map<int, int> constraint_counts_; // row -> count in current window
122 int head_ = 0;
123 int capacity_;
124
125 public:
126 explicit ConstraintTracker(int capacity)
127 : buffer_(capacity, -1)
128 , capacity_(capacity) {
129 }
130
131 void addConstraint(int row) {
132 auto it = constraint_counts_.find(row);
133 if (it != constraint_counts_.end()) {
134 constraint_counts_[row]++;
135 return;
136 }
137
138 if (buffer_[head_] != -1) { // Slot is occupied
139 constraint_counts_.erase(buffer_[head_]);
140 }
141 buffer_[head_] = row;
142 constraint_counts_[row] = 1;
143 head_ = (head_ + 1) % capacity_;
144 }
145
146 int getFrequency(int row) const {
147 auto it = constraint_counts_.find(row);
148 return (it != constraint_counts_.end()) ? it->second : 0;
149 }
150
151 void clear() {
152 constraint_counts_.clear();
153 head_ = 0;
154 }
155 };

157 public:
