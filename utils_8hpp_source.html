27// GenericPtr wrapper around std::shared_ptr or std::unique_ptr
28template<template<typename> class SmartPtr, typename T>
29class GenericPtr : public SmartPtr<T> {
30 private:
31 // Helper function to create the correct smart pointer type
32 template<typename Derived, typename... Args>
33 static SmartPtr<T> make_ptr(Args&&... args) {
34 if constexpr (std::is_same_v<SmartPtr<T>, std::shared_ptr<T>>) {
35 return std::make_shared<Derived>(std::forward<Args>(args)...);
36 } else if constexpr (std::is_same_v<SmartPtr<T>, std::unique_ptr<T>>) {
37 return std::make_unique<Derived>(std::forward<Args>(args)...);
38 } else {
39 static_assert(sizeof(SmartPtr<T>) == 0, "Unsupported smart pointer type");
40 }
41 }
42
43 public:
44 using SmartPtr<T>::SmartPtr; // Inherit constructors
45
46 template<
47 typename Derived = T,
48 typename... Args,
49 // Only enable if there are arguments AND (either T is not abstract or Derived is not T)
50 typename = std::enable_if_t<(sizeof...(Args) > 0) && (!std::is_abstract_v<T> || !std::is_same_v<Derived, T>)>>
51 GenericPtr(Args&&... args)
52 : SmartPtr<T>(make_ptr<Derived>(std::forward<Args>(args)...)) {
53 }
54
55 template<typename Derived>
56 Derived* cast() {
57 return static_cast<Derived*>(SmartPtr<T>::get());
58 }
59};
